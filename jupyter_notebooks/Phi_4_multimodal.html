Phi-4 multimodal test

Phi-4 multimodal is large language model with 5.57B parameters developed by Microsoft. It is capable of accepting text, image and audio as input. Its audio recognition performance ranked No.1 in Open ASR leaderborad.

This notebook shows a demo of Phi-4 multimodal running on Colab free T4 GPU.

# download and load the model, will take 4 minutes.
import torch
import os
import soundfile as sf
from transformers import AutoModelForCausalLM, AutoProcessor, GenerationConfig
import gradio as gr
from PIL import Image


# Define model path
model_path = "microsoft/Phi-4-multimodal-instruct"

# Load model and processor
processor = AutoProcessor.from_pretrained(model_path, trust_remote_code=True)
model = AutoModelForCausalLM.from_pretrained(
    model_path,
    device_map="cuda",
    torch_dtype="auto",
    trust_remote_code=True,
    # if you do not use Ampere or later GPUs, change attention to "eager"
    _attn_implementation="eager"#'flash_attention_2',
).cuda()
generation_config = GenerationConfig.from_pretrained(model_path)
# Define prompt structure
user_prompt = '<|user|>'
assistant_prompt = '<|assistant|>'
prompt_suffix = '<|end|>'



def process_image_and_generate(user_message, image):
    #print("\n--- IMAGE PROCESSING ---")
    prompt = f'{user_prompt}<|image_1|>{user_message}{prompt_suffix}{assistant_prompt}'
    #print(f'>>> Prompt\n{prompt}')

    inputs = processor(text=prompt, images=image, return_tensors='pt').to('cuda:0')

    # Generate response
    generate_ids = model.generate(
        **inputs,
        max_new_tokens=1000,
        generation_config=generation_config,
    )
    generate_ids = generate_ids[:, inputs['input_ids'].shape[1]:]
    response = processor.batch_decode(
        generate_ids, skip_special_tokens=True, clean_up_tokenization_spaces=False
    )[0]
    return response


def process_audio_and_generate(speech_prompt, audio_path):
    prompt = f'{user_prompt}<|audio_1|>{speech_prompt}{prompt_suffix}{assistant_prompt}'
    #print(f'>>> Prompt\n{prompt}')

    # Downlowd and open audio file
    audio, samplerate = sf.read(audio_path)

    # Process with the model
    inputs = processor(text=prompt, audios=[(audio, samplerate)], return_tensors='pt').to('cuda:0')

    generate_ids = model.generate(
        **inputs,
        max_new_tokens=1000,
        generation_config=generation_config,
    )
    generate_ids = generate_ids[:, inputs['input_ids'].shape[1]:]
    response = processor.batch_decode(
        generate_ids, skip_special_tokens=True, clean_up_tokenization_spaces=False
    )[0]
    return response


with gr.Blocks() as demo:
    gr.Markdown("# Image/Audio Processing with Message")
    with gr.Tab("Image"):
        image_input = gr.Image(type="pil", label="Upload an image")
        user_message_image = gr.Textbox(label="User Prompt")
        image_output = gr.Textbox(label="Output")
        image_button = gr.Button("Process Image")
        image_button.click(fn=process_image_and_generate, inputs=[user_message_image, image_input], outputs=image_output)

    with gr.Tab("Audio"):
        audio_input = gr.Audio(type="filepath", label="Upload Audio")
        user_message_audio = gr.Textbox(label="User Prompt")
        audio_output = gr.Textbox(label="Output")
        audio_button = gr.Button("Process Audio")
        audio_button.click(fn=process_audio_and_generate, inputs=[user_message_audio, audio_input], outputs=audio_output),
